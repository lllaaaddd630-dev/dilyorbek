        // Snow animation
        function initSnow() {
            const snowContainer = document.getElementById('snow-container');
            const snowflakeCount = 100;
            
            for (let i = 0; i < snowflakeCount; i++) {
                createSnowflake();
            }
            
            function createSnowflake() {
                const snowflake = document.createElement('div');
                snowflake.classList.add('snowflake');
                
                const size = 2 + Math.random() * 8;
                snowflake.style.width = `${size}px`;
                snowflake.style.height = `${size}px`;
                
                const startX = Math.random() * 100;
                snowflake.style.left = `${startX}vw`;
                
                const opacity = 0.3 + Math.random() * 0.7;
                snowflake.style.opacity = opacity;
                
                const blur = 0.5 + Math.random() * 1.5;
                snowflake.style.filter = `blur(${blur}px)`;
                
                const duration = 8 + Math.random() * 12;
                const delay = Math.random() * 5;
                snowflake.style.animationDuration = `${duration}s`;
                snowflake.style.animationDelay = `${delay}s`;
                
                snowContainer.appendChild(snowflake);
                
                setTimeout(() => {
                    if (snowflake.parentNode) {
                        snowflake.remove();
                        createSnowflake();
                    }
                }, (duration + delay) * 1000);
            }
        }

        // Ice crystals animation
        function initIceCrystals() {

            function createIceCrystal() {
                const crystal = document.createElement('div');
                crystal.classList.add('ice-crystal');
                crystal.textContent = crystals[Math.floor(Math.random() * crystals.length)];
                
                const leftPos = 5 + Math.random() * 90;
                crystal.style.left = `${leftPos}%`;
                
                const duration = 20 + Math.random() * 20;
                const delay = Math.random() * 15;
                crystal.style.animation = `floatIce ${duration}s linear ${delay}s infinite`;
                
                iceContainer.appendChild(crystal);
            }
        }

        // Birthday countdown
        function updateCountdown() {
            const now = new Date();
            const currentYear = now.getFullYear();
            
            // Set birthday to December 8th
            let nextBirthday = new Date(currentYear, 11, 8); // December is month 11
            
            // If birthday has already passed this year, set to next year
            if (now > nextBirthday) {
                nextBirthday = new Date(currentYear + 1, 11, 8);
            }
            
            const timeDiff = nextBirthday - now;
            
            // Calculate days, hours, minutes, seconds
            const days = Math.floor(timeDiff / (1000 * 60 * 60 * 24));
            const hours = Math.floor((timeDiff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((timeDiff % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((timeDiff % (1000 * 60)) / 1000);
            
            // Update display
            document.getElementById('days').textContent = days;
            document.getElementById('hours').textContent = hours.toString().padStart(2, '0');
            document.getElementById('minutes').textContent = minutes.toString().padStart(2, '0');
            document.getElementById('seconds').textContent = seconds.toString().padStart(2, '0');
            
            // Update total days
            document.getElementById('total-days').textContent = `Keyingi tug'ilgan kunigacha: ${days} kun`;
            
            // Update next birthday date
            const options = { year: 'numeric', month: 'long', day: 'numeric' };
            document.getElementById('next-birthday').textContent = 
                `Keyingi tug'ilgan kun: ${nextBirthday.toLocaleDateString('uz-UZ', options)}`;
        }

        // Initialize everything when page loads
        document.addEventListener('DOMContentLoaded', function() {
            initSnow();
            initIceCrystals();
            updateCountdown();
            setInterval(updateCountdown, 1000);
        });
